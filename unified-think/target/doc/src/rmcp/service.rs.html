217/// A handle to a remote request
218///
219/// You can cancel it by call [`RequestHandle::cancel`] with a reason,
220///
221/// or wait for response by call [`RequestHandle::await_response`]
222#[derive(Debug)]
223pub struct RequestHandle<R: ServiceRole> {
224    pub rx: tokio::sync::oneshot::Receiver<Result<R::PeerResp, ServiceError>>,
225    pub options: PeerRequestOptions,
226    pub peer: Peer<R>,
227    pub id: RequestId,
228}
229
230impl<R: ServiceRole> RequestHandle<R> {
231    pub const REQUEST_TIMEOUT_REASON: &str = "request timeout";
232    pub async fn await_response(self) -> Result<R::PeerResp, ServiceError> {
233        if let Some(timeout) = self.options.timeout {
234            let timeout_result = tokio::time::timeout(timeout, async move {
235                self.rx
236                    .await
237                    .map_err(|_e| ServiceError::Transport(std::io::Error::other("disconnected")))?
238            })
239            .await;
240            match timeout_result {
241                Ok(response) => response,
242                Err(_) => {
243                    let error = Err(ServiceError::Timeout { timeout });
244                    // cancel this request
245                    let notification = CancelledNotification {
246                        params: CancelledNotificationParam {
247                            request_id: self.id,
248                            reason: Some(Self::REQUEST_TIMEOUT_REASON.to_owned()),
249                        },
250                        method: crate::model::CancelledNotificationMethod,
251                    };
252                    let _ = self.peer.send_notification(notification.into()).await;
253                    error
254                }
255            }
256        } else {
257            self.rx
258                .await
259                .map_err(|_e| ServiceError::Transport(std::io::Error::other("disconnected")))?
260        }
261    }
262
263    /// Cancel this request
264    pub async fn cancel(self, reason: Option<String>) -> Result<(), ServiceError> {
265        let notification = CancelledNotification {
266            params: CancelledNotificationParam {
267                request_id: self.id,
268                reason,
269            },
270            method: crate::model::CancelledNotificationMethod,
271        };
272        self.peer.send_notification(notification.into()).await?;
273        Ok(())
274    }
275}

321impl<R: ServiceRole> Peer<R> {
322    const CLIENT_CHANNEL_BUFFER_SIZE: usize = 1024;
323    pub fn new(
324        request_id_provider: Arc<dyn RequestIdProvider>,
325        peer_info: R::PeerInfo,
326    ) -> (Peer<R>, ProxyOutbound<R>) {
327        let (tx, rx) = mpsc::channel(Self::CLIENT_CHANNEL_BUFFER_SIZE);
328        (
329            Self {
330                tx,
331                request_id_provider,
332                info: peer_info.into(),
333            },
334            rx,
335        )
336    }
337    pub async fn send_notification(&self, notification: R::Not) -> Result<(), ServiceError> {
338        let (responder, receiver) = tokio::sync::oneshot::channel();
339        self.tx
340            .send(PeerSinkMessage::Notification(notification, responder))
341            .await
342            .map_err(|_m| ServiceError::Transport(std::io::Error::other("disconnected")))?;
343        receiver
344            .await
345            .map_err(|_e| ServiceError::Transport(std::io::Error::other("disconnected")))?
346    }
347    pub async fn send_request(&self, request: R::Req) -> Result<R::PeerResp, ServiceError> {
348        self.send_cancellable_request(request, PeerRequestOptions::no_options())
349            .await?
350            .await_response()
351            .await
352    }
353    pub async fn send_cancellable_request(
354        &self,
355        request: R::Req,
356        options: PeerRequestOptions,
357    ) -> Result<RequestHandle<R>, ServiceError> {
358        let id = self.request_id_provider.next_request_id();
359        let (responder, receiver) = tokio::sync::oneshot::channel();
360        self.tx
361            .send(PeerSinkMessage::Request(request, id.clone(), responder))
362            .await
363            .map_err(|_m| ServiceError::Transport(std::io::Error::other("disconnected")))?;
364        Ok(RequestHandle {
365            id,
366            rx: receiver,
367            options,
368            peer: self.clone(),
369        })
370    }
371    pub fn peer_info(&self) -> &R::PeerInfo {
372        &self.info
373    }
374}

457#[instrument(skip_all)]
458async fn serve_inner<R, S, T, E, A>(
459    mut service: S,
460    transport: T,
461    peer_info: R::PeerInfo,
462    id_provider: Arc<AtomicU32RequestIdProvider>,
463    ct: CancellationToken,
464) -> Result<RunningService<R, S>, E>
465where
466    R: ServiceRole,
467    S: Service<R>,
468    T: IntoTransport<R, E, A>,
469    E: std::error::Error + Send + Sync + 'static,
470{
471    use futures::{SinkExt, StreamExt};
472    const SINK_PROXY_BUFFER_SIZE: usize = 64;
473    let (sink_proxy_tx, mut sink_proxy_rx) = tokio::sync::mpsc::channel::<
474        Message<<R as ServiceRole>::Req, <R as ServiceRole>::Resp, <R as ServiceRole>::Not>,
475    >(SINK_PROXY_BUFFER_SIZE);
476
477    if R::IS_CLIENT {
478        tracing::info!(?peer_info, "Service initialized as client");
479    } else {
480        tracing::info!(?peer_info, "Service initialized as server");
481    }
482
483    let (peer, mut peer_proxy) = <Peer<R>>::new(id_provider, peer_info);
484    service.set_peer(peer.clone());
485    let mut local_responder_pool = HashMap::new();
486    let mut local_ct_pool = HashMap::<RequestId, CancellationToken>::new();
487    let shared_service = Arc::new(service);
488    // for return
489    let service = shared_service.clone();
490
491    // let message_sink = tokio::sync::
492    // let mut stream = std::pin::pin!(stream);
493    let serve_loop_ct = ct.child_token();
494    let peer_return: Peer<R> = peer.clone();
495    let handle = tokio::spawn(async move {
496        let (mut sink, mut stream) = transport.into_transport();
497        let mut sink = std::pin::pin!(sink);
498        let mut stream = std::pin::pin!(stream);
499        #[derive(Debug)]
500        enum Event<P, R, T> {
501            ProxyMessage(P),
502            PeerMessage(R),
503            ToSink(T),
504        }
505        let quit_reason = loop {
506            let evt = tokio::select! {
507                m = sink_proxy_rx.recv() => {
508                    if let Some(m) = m {
509                        Event::ToSink(m)
510                    } else {
511                        continue
512                    }
513                }
514                m = stream.next() => {
515                    if let Some(m) = m {
516                        Event::PeerMessage(m.into_message())
517                    } else {
518                        // input stream closed
519                        tracing::info!("input stream terminated");
520                        break QuitReason::Closed
521                    }
522                }
523                m = peer_proxy.recv() => {
524                    if let Some(m) = m {
525                        Event::ProxyMessage(m)
526                    } else {
527                        continue
528                    }
529                }
530                _ = serve_loop_ct.cancelled() => {
531                    tracing::info!("task cancelled");
532                    break QuitReason::Cancelled
533                }
534            };
535            tracing::debug!(?evt, "new event");
536            match evt {
537                // response and error
538                Event::ToSink(e) => {
539                    if let Some(id) = match &e {
540                        Message::Response(_, id) => Some(id),
541                        Message::Error(_, id) => Some(id),
542                        _ => None,
543                    } {
544                        if let Some(ct) = local_ct_pool.remove(id) {
545                            ct.cancel();
546                        }
547                        let send_result = sink.send(e.into_json_rpc_message()).await;
548                        if let Err(error) = send_result {
549                            tracing::error!(%error, "fail to response message");
550                        }
551                    }
552                }
553                Event::ProxyMessage(PeerSinkMessage::Request(request, id, responder)) => {
554                    local_responder_pool.insert(id.clone(), responder);
555                    let send_result = sink
556                        .send(Message::Request(request, id.clone()).into_json_rpc_message())
557                        .await;
558                    if let Err(e) = send_result {
559                        if let Some(responder) = local_responder_pool.remove(&id) {
560                            let _ = responder
561                                .send(Err(ServiceError::Transport(std::io::Error::other(e))));
562                        }
563                    }
564                }
565                Event::ProxyMessage(PeerSinkMessage::Notification(notification, responder)) => {
566                    // catch cancellation notification
567                    let mut cancellation_param = None;
568                    let notification = match notification.try_into() {
569                        Ok::<CancelledNotification, _>(cancelled) => {
570                            cancellation_param.replace(cancelled.params.clone());
571                            cancelled.into()
572                        }
573                        Err(notification) => notification,
574                    };
575                    let send_result = sink
576                        .send(Message::Notification(notification).into_json_rpc_message())
577                        .await;
578                    if let Err(e) = send_result {
579                        let _ =
580                            responder.send(Err(ServiceError::Transport(std::io::Error::other(e))));
581                    }
582                    if let Some(param) = cancellation_param {
583                        if let Some(responder) = local_responder_pool.remove(&param.request_id) {
584                            tracing::info!(id = %param.request_id, reason = param.reason, "cancelled");
585                            let _response_result = responder.send(Err(ServiceError::Cancelled {
586                                reason: param.reason.clone(),
587                            }));
588                        }
589                    }
590                }
591                Event::PeerMessage(Message::Request(request, id)) => {
592                    tracing::info!(%id, ?request, "received request");
593                    {
594                        let service = shared_service.clone();
595                        let sink = sink_proxy_tx.clone();
596                        let request_ct = serve_loop_ct.child_token();
597                        let context_ct = request_ct.child_token();
598                        local_ct_pool.insert(id.clone(), request_ct);
599                        let context = RequestContext {
600                            ct: context_ct,
601                            id: id.clone(),
602                            peer: peer.clone(),
603                        };
604                        tokio::spawn(async move {
605                            let result = service.handle_request(request, context).await;
606                            let response = match result {
607                                Ok(result) => {
608                                    tracing::info!(%id, ?result, "response message");
609                                    Message::Response(result, id)
610                                }
611                                Err(error) => {
612                                    tracing::warn!(%id, ?error, "response error");
613                                    Message::Error(error, id)
614                                }
615                            };
616                            let _send_result = sink.send(response).await;
617                        });
618                    }
619                }
620                Event::PeerMessage(Message::Notification(notification)) => {
621                    tracing::info!(?notification, "received notification");
622                    // catch cancelled notification
623                    let notification = match notification.try_into() {
624                        Ok::<CancelledNotification, _>(cancelled) => {
625                            if let Some(ct) = local_ct_pool.remove(&cancelled.params.request_id) {
626                                tracing::info!(id = %cancelled.params.request_id, reason = cancelled.params.reason, "cancelled");
627                                ct.cancel();
628                            }
629                            cancelled.into()
630                        }
631                        Err(notification) => notification,
632                    };
633                    {
634                        let service = shared_service.clone();
635                        tokio::spawn(async move {
636                            let result = service.handle_notification(notification).await;
637                            if let Err(error) = result {
638                                tracing::warn!(%error, "Error sending notification");
639                            }
640                        });
641                    }
642                }
643                Event::PeerMessage(Message::Response(result, id)) => {
644                    if let Some(responder) = local_responder_pool.remove(&id) {
645                        let response_result = responder.send(Ok(result));
646                        if let Err(_error) = response_result {
647                            tracing::warn!(%id, "Error sending response");
648                        }
649                    }
650                }
651                Event::PeerMessage(Message::Error(error, id)) => {
652                    if let Some(responder) = local_responder_pool.remove(&id) {
653                        let _response_result = responder.send(Err(ServiceError::McpError(error)));
654                        if let Err(_error) = _response_result {
655                            tracing::warn!(%id, "Error sending response");
656                        }
657                    }
658                }
659            }
660        };
661        let sink_close_result = sink.close().await;
662        if let Err(e) = sink_close_result {
663            tracing::error!(%e, "fail to close sink");
664        }
665        tracing::info!(?quit_reason, "serve finished");
666        quit_reason
667    });
668    Ok(RunningService {
669        service,
670        peer: peer_return,
671        handle,
672        ct,
673    })
